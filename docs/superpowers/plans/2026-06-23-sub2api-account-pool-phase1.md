# sub2api Account Pool Phase 1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build the Phase 1 account-pool foundation: persistent models, encrypted secret storage, admin CRUD APIs, draft channel bindings, and a minimal admin UI that cannot send real traffic through account-pool channels yet.

**Architecture:** Keep account pools as a separate domain from ordinary Wynth channels. Phase 1 stores and manages pools, accounts, proxies, and draft bindings; relay runtime stays unchanged, and bound channels are kept non-selectable by forcing them to remain manually disabled. Sensitive account, token, and proxy values are stored through a common encrypted envelope backed by `common.CryptoSecret`.

**Tech Stack:** Go 1.22+, Gin, GORM v2, SQLite/MySQL/PostgreSQL-compatible schema, testify, React 19, TypeScript, TanStack Router/Query, Base UI, Tailwind CSS, Bun.

---

## Scope

This plan implements only Phase 1 from [the design spec](../specs/2026-06-23-sub2api-account-pool-design.md).

It intentionally does not implement:

- relay-time account selection;
- OpenAI OAuth refresh;
- account-level retry;
- ChatGPT reverse proxy protocol adaptation;
- account metrics aggregation into auto-priority.

The Phase 1 success condition is: an admin can create account pools, proxies, accounts, and draft bindings; secrets are encrypted at rest and masked in responses; and any channel bound to an account pool remains unavailable to the existing outer channel scheduler.

Create `common/secret_envelope.go`:

```go
package common

import (
	"crypto/aes"
	"crypto/cipher"
	"crypto/rand"
	"crypto/sha256"
	"encoding/base64"
	"fmt"
	"io"
)

const secretEnvelopeAlgAESGCM = "AES-256-GCM"

type SecretEnvelope struct {
	Version    int    `json:"v"`
	Algorithm  string `json:"alg"`
	Nonce      string `json:"nonce"`
	Ciphertext string `json:"ciphertext"`
}

func EncryptSecretString(plaintext string) (string, error) {
	if plaintext == "" {
		return "", nil
	}
	gcm, err := secretEnvelopeGCM()
	if err != nil {
		return "", err
	}
	nonce := make([]byte, gcm.NonceSize())
	if _, err := io.ReadFull(rand.Reader, nonce); err != nil {
		return "", err
	}
	ciphertext := gcm.Seal(nil, nonce, []byte(plaintext), nil)
	payload, err := Marshal(SecretEnvelope{
		Version:    1,
		Algorithm:  secretEnvelopeAlgAESGCM,
		Nonce:      base64.StdEncoding.EncodeToString(nonce),
		Ciphertext: base64.StdEncoding.EncodeToString(ciphertext),
	})
	if err != nil {
		return "", err
	}
	return string(payload), nil
}

func DecryptSecretString(envelope string) (string, error) {
	if envelope == "" {
		return "", nil
	}
	var payload SecretEnvelope
	if err := UnmarshalJsonStr(envelope, &payload); err != nil {
		return "", err
	}
	if payload.Version != 1 || payload.Algorithm != secretEnvelopeAlgAESGCM {
		return "", fmt.Errorf("unsupported secret envelope")
	}
	nonce, err := base64.StdEncoding.DecodeString(payload.Nonce)
	if err != nil {
		return "", err
	}
	ciphertext, err := base64.StdEncoding.DecodeString(payload.Ciphertext)
	if err != nil {
		return "", err
	}
	gcm, err := secretEnvelopeGCM()
	if err != nil {
		return "", err
	}
	plaintext, err := gcm.Open(nil, nonce, ciphertext, nil)
	if err != nil {
		return "", err
	}
	return string(plaintext), nil
}

func secretEnvelopeGCM() (cipher.AEAD, error) {
	if CryptoSecret == "" {
		return nil, fmt.Errorf("crypto secret is empty")
	}
	sum := sha256.Sum256([]byte("wynth-account-pool:" + CryptoSecret))
	block, err := aes.NewCipher(sum[:])
	if err != nil {
		return nil, err
	}
	return cipher.NewGCM(block)
}
```

Use project JSON wrappers (`Marshal`, `UnmarshalJsonStr`), not direct `encoding/json`.

```go
package service

import (
	"strings"

	"github.com/QuantumNous/new-api/common"
)

const (
	AccountPoolCredentialTypeAPIKey = "api_key"
	AccountPoolCredentialTypeOAuth  = "oauth"
)

type AccountPoolCredentialConfig struct {
	Type         string `json:"type"`
	APIKey       string `json:"api_key,omitempty"`
	Email        string `json:"email,omitempty"`
	RefreshToken string `json:"refresh_token,omitempty"`
}

type AccountPoolTokenState struct {
	AccessToken  string `json:"access_token,omitempty"`
	RefreshToken string `json:"refresh_token,omitempty"`
	ExpiresAt    int64  `json:"expires_at,omitempty"`
	Version      int64  `json:"version"`
}

type AccountPoolProxyAuthConfig struct {
	Username string `json:"username,omitempty"`
	Password string `json:"password,omitempty"`
}

type AccountPoolModelPolicy struct {
	Strategy    string   `json:"strategy"`
	FixedModels []string `json:"fixed_models,omitempty"`
}

type AccountPoolAccountFilterConfig struct {
	AccountIDs []int `json:"account_ids,omitempty"`
}

func (s AccountPoolTokenState) NextVersion() AccountPoolTokenState {
	s.Version++
	return s
}

func EncryptAccountPoolCredentialConfig(config AccountPoolCredentialConfig) (string, error) {
	payload, err := common.Marshal(config)
	if err != nil {
		return "", err
	}
	return common.EncryptSecretString(string(payload))
}

func DecryptAccountPoolCredentialConfig(encrypted string) (AccountPoolCredentialConfig, error) {
	var config AccountPoolCredentialConfig
	plaintext, err := common.DecryptSecretString(encrypted)
	if err != nil || plaintext == "" {
		return config, err
	}
	err = common.UnmarshalJsonStr(plaintext, &config)
	return config, err
}

func MaskAccountPoolSecretValue(value string) string {
	value = strings.TrimSpace(value)
	if value == "" {
		return ""
	}
	if len(value) <= 8 {
		return "***"
	}
	return value[:4] + "..." + value[len(value)-4:]
}
```

```ts
    values.credentialType === 'api_key'
      ? { type: 'api_key', api_key: values.apiKey }
      : { type: 'oauth', email: values.email, refresh_token: values.refreshToken }
  return {
    name: values.name.trim(),
    account_identifier: values.accountIdentifier.trim(),
    credential,
    priority: values.priority,
    weight: values.weight,
    max_concurrency: values.maxConcurrency,
    proxy_id: values.proxyID,
    supported_models: normalizeModelListText(values.supportedModelsText),
    model_mapping: values.modelMapping,
  }
}

export function maskSecretPreview(value: string): string {
  const trimmed = value.trim()
  if (!trimmed) return ''
  if (trimmed.length <= 8) return '***'
  return `${trimmed.slice(0, 4)}...${trimmed.slice(-4)}`
}
```
